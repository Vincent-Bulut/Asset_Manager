import pandas as pd
import json

# Charger le fichier Excel en spécifiant que la première ligne est l'en-tête
excel_file_path = 'votre_fichier.xlsx'
df = pd.read_excel(excel_file_path, header=0)

# Extraire les noms des colonnes (le header) sous forme de liste
header_list = df.columns.tolist()

# Sauvegarder cette liste dans un fichier JSON
json_file_path = 'header.json'
with open(json_file_path, 'w', encoding='utf-8') as json_file:
    json.dump(header_list, json_file, ensure_ascii=False, indent=4)

print("Le header a été sauvegardé dans le fichier JSON.")
