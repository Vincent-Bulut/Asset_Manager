
<script>
  import axios from 'axios';

  async function downloadFile(id_folder, s3_key) {
    try {
      // Construire l'URL
      const url = `/folder/${id_folder}/files/${s3_key}`;

      // Faire une requête GET avec Axios
      const response = await axios.get(url, {
        responseType: 'blob', // Important : indique qu'on attend un fichier en réponse
      });

      // Vérifiez que la réponse est correcte
      if (response.status !== 200) {
        throw new Error(`Erreur HTTP: ${response.status} - ${response.statusText}`);
      }

      // Récupérer le fichier (Blob)
      const blob = new Blob([response.data]);

      // Créer une URL temporaire pour le fichier
      const blobUrl = URL.createObjectURL(blob);

      // Récupérer le nom du fichier depuis les headers
      const contentDisposition = response.headers['content-disposition'];
      let filename = 'downloaded_file'; // Nom par défaut
      if (contentDisposition) {
        const matches = contentDisposition.match(/filename="(.+)"/);
        if (matches && matches[1]) {
          filename = matches[1];
        }
      }

      // Créer un lien pour télécharger le fichier
      const link = document.createElement('a');
      link.href = blobUrl;
      link.download = filename; // Nom du fichier
      document.body.appendChild(link);
      link.click(); // Simuler un clic pour démarrer le téléchargement
      document.body.removeChild(link);

      // Libérer l'URL pour éviter des fuites mémoire
      URL.revokeObjectURL(blobUrl);
    } catch (error) {
      console.error('Erreur lors du téléchargement:', error);
      alert('Impossible de télécharger le fichier.');
    }
  }
</script>
