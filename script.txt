
import pandas as pd

def check_dataframes_shape(*dataframes):
    if not dataframes:
        raise ValueError("No DataFrames provided")

    # Get the shape of the first dataframe
    first_shape = dataframes[0].shape

    # Check if all DataFrames have the same shape
    for df in dataframes:
        if df.shape != first_shape:
            raise ValueError("Not all DataFrames have the same shape")

    print("All DataFrames have the same shape")

# Example usage
df1 = pd.DataFrame({'A': [1, 2, 3], 'B': [4, 5, 6]})
df2 = pd.DataFrame({'A': [7, 8, 9], 'B': [10, 11, 12]})
df3 = pd.DataFrame({'A': [13, 14, 15], 'B': [16, 17, 18]})

check_dataframes_shape(df1, df2, df3)  # This should print "All DataFrames have the same shape"
