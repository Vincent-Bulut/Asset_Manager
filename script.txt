
<script>
  let users = [
    { id: 'ut3k8g', firstName: 'vincent', lastName: 'bulut', role: 'admin', disabled: false }
    // ... vous pouvez ajouter plus d'utilisateurs ici
  ];
  
  let searchQuery = '';

  // Cette fonction retourne les utilisateurs filtrés en fonction de la requête de recherche
  $: filteredUsers = users.filter(user => {
    return user.firstName.toLowerCase().includes(searchQuery.toLowerCase()) || 
           user.lastName.toLowerCase().includes(searchQuery.toLowerCase()) ||
           user.id.toLowerCase().includes(searchQuery.toLowerCase());
  });
</script>

<!-- Barre de recherche -->
<input type="text" bind:value={searchQuery} placeholder="Search users..." />

<!-- Tableau des utilisateurs -->
<table>
  <tr>
    <th>Id</th>
    <th>First Name</th>
    <th>Last Name</th>
    <th>Role</th>
    <th>Disabled</th>
    <th>Action</th>
  </tr>
  {#each filteredUsers as user}
    <tr>
      <td>{user.id}</td>
      <td>{user.firstName}</td>
      <td>{user.lastName}</td>
      <td>{user.role}</td>
      <td>{user.disabled.toString()}</td>
      <td><button>Update</button></td>
    </tr>
  {/each}
</table>
