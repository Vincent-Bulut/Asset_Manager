
<script>
  let clients = ['Client 1', 'Client 2', 'Client 3'];
  let addedClients = ['client 4', 'client 5'];

  function moveToAdded(client) {
    addedClients = [...addedClients, client];
    clients = clients.filter(c => c !== client);
  }

  function moveToClients(client) {
    clients = [...clients, client];
    addedClients = addedClients.filter(c => c !== client);
  }
</script>

<style>
  .list {
    height: 150px; /* Votre hauteur fixe */
    overflow-y: auto;
    border: 1px solid black;
  }
  .list-header {
    position: sticky;
    top: 0;
    background: white;
  }
  .button-container {
    display: flex;
    flex-direction: column;
    justify-content: center;
    padding: 10px;
  }
  button {
    margin: 5px;
  }
</style>

<div class="list">
  <div class="list-header">List Clients 1</div>
  {#each clients as client}
    <div>{client}</div>
  {/each}
</div>

<div class="button-container">
  <button on:click={() => moveToAdded(clients[0])}>&gt;&gt;</button>
  <button on:click={() => moveToClients(addedClients[0])}>&lt;&lt;</button>
</div>

<div class="list">
  <div class="list-header">Clients ajout√©s</div>
  {#each addedClients as client}
    <div>{client}</div>
  {/each}
</div>
