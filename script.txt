
stages:
  - dependencies-report

# Étape pour générer un rapport des dépendances Python
python_dependencies_report:
  stage: dependencies-report
  image: python:3.9
  script:
    - pip install -r requirements.txt  # Installer les dépendances
    - pip freeze > python-dependencies.txt  # Générer le rapport des dépendances Python
  artifacts:
    paths:
      - python-dependencies.txt

# Étape pour générer un rapport des dépendances JavaScript
javascript_dependencies_report:
  stage: dependencies-report
  image: node:16
  script:
    - npm install  # Installer les dépendances
    - npm list --depth=0 > javascript-dependencies.txt  # Générer le rapport des dépendances JS
  artifacts:
    paths:
      - javascript-dependencies.txt
