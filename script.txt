    <button onclick="genererXML()">Générer XML</button>

    <!-- Affichage XML -->
    <h2>Sortie XML :</h2>
    <pre id="outputXML"></pre>

    <script>
        function genererXML() {
            const content = document.getElementById("content");
            const elements = content.querySelectorAll("p"); // Sélectionne toutes les balises <p>

            let xmlOutput = `<données>\n`;

            // Parcourir chaque <p> pour récupérer son texte et l'élément suivant (input, select, etc.)
            elements.forEach(p => {
                const tagName = p.textContent.trim(); // Le nom de la balise XML (texte dans <p>)
                const nextElement = p.nextElementSibling; // L'élément suivant <p> (input ou select)

                if (nextElement && (nextElement.tagName === "INPUT" || nextElement.tagName === "SELECT")) {
                    const value = nextElement.value; // La valeur saisie ou sélectionnée
                    xmlOutput += `    <${tagName}>${value}</${tagName}>\n`;
                }
            });

            xmlOutput += `</données>`;

            // Affichage dans la zone de sortie
            document.getElementById("outputXML").textContent = xmlOutput;
        }
    </script>
