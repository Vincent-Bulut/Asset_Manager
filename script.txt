
Sub CopierFeuille()
    Dim SourceWorkbook As Workbook
    Dim DestinationWorkbook As Workbook
    Dim SourceSheet As Worksheet
    Dim DestinationSheet As Worksheet

    ' Chemin du classeur source
    Dim SourcePath As String
    SourcePath = "C:\Chemin\Vers\Votre\ClasseurSource.xlsx"
    
    ' Chemin du classeur de destination
    Dim DestinationPath As String
    DestinationPath = "C:\Chemin\Vers\Votre\ClasseurDestination.xlsx"
    
    ' Ouvrir le classeur source
    Set SourceWorkbook = Workbooks.Open(SourcePath)
    
    ' Définir la feuille à copier
    Set SourceSheet = SourceWorkbook.Sheets("NomDeLaFeuilleSource")
    
    ' Ouvrir le classeur de destination
    Set DestinationWorkbook = Workbooks.Open(DestinationPath)
    
    ' Copier la feuille source dans le classeur de destination
    SourceSheet.Copy After:=DestinationWorkbook.Sheets(DestinationWorkbook.Sheets.Count)
    
    ' Optionnel : Renommer la feuille copiée
    Set DestinationSheet = DestinationWorkbook.Sheets(DestinationWorkbook.Sheets.Count)
    DestinationSheet.Name = "NomDeLaNouvelleFeuille"
    
    ' Fermer le classeur source sans sauvegarder les modifications
    SourceWorkbook.Close SaveChanges:=False
    
    ' Sauvegarder et fermer le classeur de destination
    DestinationWorkbook.Close SaveChanges:=True
    
    MsgBox "La feuille a été copiée avec succès !"
End Sub
