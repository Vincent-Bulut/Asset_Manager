
Sub GetFileSize()
    Dim filePath As String
    Dim fileSize As Double
    
    ' Obtenir le chemin complet du fichier actuel
    filePath = ThisWorkbook.FullName
    
    ' Vérifier si le fichier existe
    If Dir(filePath) <> "" Then
        ' Utiliser le FileSystemObject pour obtenir la taille du fichier
        Dim fso As Object
        Set fso = CreateObject("Scripting.FileSystemObject")
        
        ' Récupérer la taille du fichier en octets
        fileSize = fso.GetFile(filePath).Size
        
        ' Afficher la taille du fichier en octets
        MsgBox "La taille du fichier est de " & fileSize & " octets."
    Else
        MsgBox "Le fichier n'existe pas."
    End If
End Sub
