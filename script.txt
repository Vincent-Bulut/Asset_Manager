
const stack = []; // Création de la pile avec un tableau

// Empiler des éléments
stack.push(10);
stack.push(20);
stack.push(30);

// Récupérer les éléments successivement (Dépilement)
console.log("1er élément retiré :", stack.pop()); // 30
console.log("2ème élément retiré :", stack.pop()); // 20
console.log("3ème élément retiré :", stack.pop()); // 10

// Gestion de la pile vide
console.log("Pile vide :", stack.pop()); // undefined
