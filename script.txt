from sqlalchemy.orm import sessionmaker
from your_model_file import BusinessManager, Traders, engine

# Créer une session
Session = sessionmaker(bind=engine)
session = Session()

# Mettre à jour le statut des BusinessManager
business_managers = session.query(BusinessManager).all()
for manager in business_managers:
    manager.publication_status = 'nouveau_statut'

# Mettre à jour le statut des Traders
traders = session.query(Traders).all()
for trader in traders:
    trader.publication_status = 'nouveau_statut'

# Commit les changements
session.commit()
