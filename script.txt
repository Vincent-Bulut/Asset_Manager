
<script>
  let clients = ['client 1', 'client 2', 'client 3', 'client 4', 'client 5', 'client 6', 'client 7'];
  let addedClients = ['client 19', 'client 20'];
  let selectedClient = '';
  let selectedAddedClient = '';

  function addClient() {
    if (selectedClient && !addedClients.includes(selectedClient)) {
      addedClients = [...addedClients, selectedClient];
      clients = clients.filter(c => c !== selectedClient);
      selectedClient = '';
    }
  }

  function removeClient() {
    if (selectedAddedClient && !clients.includes(selectedAddedClient)) {
      clients = [...clients, selectedAddedClient];
      addedClients = addedClients.filter(c => c !== selectedAddedClient);
      selectedAddedClient = '';
    }
  }
</script>

<style>
  .container {
    display: flex;
    align-items: start;
    gap: 10px;
  }
  .list {
    width: 200px; /* Adjust width as necessary */
    height: 200px; /* Adjust height as necessary */
    border: 1px solid black;
    overflow-y: auto;
  }
  .button-container {
    display: flex;
    flex-direction: column;
    justify-content: center;
    gap: 5px; /* Adjust gap as necessary */
  }
  button {
    width: 50px; /* Adjust width as necessary */
  }
</style>

<div class="container">
  <div>
    <div>List Clients 1</div>
    <div class="list">
      {#each clients as client}
        <div on:click={() => selectedClient = client}>{client}</div>
      {/each}
    </div>
  </div>

  <div class="button-container">
    <button on:click={addClient}>&gt;&gt;</button>
    <button on:click={removeClient}>&lt;&lt;</button>
  </div>

  <div>
    <div>Clients ajout√©s</div>
    <div class="list">
      {#each addedClients as client}
        <div on:click={() => selectedAddedClient = client}>{client}</div>
      {/each}
    </div>
  </div>
</div>
